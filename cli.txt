import { Command } from 'commander';



export class CLI {
  getVersion() {
    const { version } = require('package.json');
    
    return version;
  }
  
  createConfig() {
  
  }
  
  run() {
    const program = new Command();
    
    // sets up the CLI version
    program.version(this.getVersion());
    
    program
      .command('init')  
      .description('Creates a default trun.json config file')
      .action(this.createConfig);
      
    program
      .option('-c, --config <type>', 'Specify a config file for the test runner', 'trun.json')
  }
}
